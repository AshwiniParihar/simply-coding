  <!-- Unordered Lists -->
  <h2 id="ingredients">Ingredients</h2>
  <ul style="list-style-type: square; padding-left: 20px; color: red">
    <li>2 cups of flour</li>
    <li>1 cup of warm water</li>
    <li>
      1 packet of yeast
      <ul style="list-style-type: circle; margin-left: 20px">
        <li>Instant yeast</li>
        <li>Dry active yeast</li>
      </ul>
    </li>
    <li>1 tablespoon of olive oil</li>
    <li>1 teaspoon of salt</li>
    <li>Cheese and toppings of your choice</li>
  </ul>

  <!-- Ordered Lists -->
  <h2 id="instructions">Instructions</h2>
  <ol style="list-style-type: upper-roman; padding-left: 20px">
    <li>
      Prepare the Dough
      <ol style="list-style-type: lower-alpha; margin-left: 20px">
        <li>Combine flour, yeast, and salt in a bowl.</li>
        <li>Add warm water and olive oil.</li>
        <li>Mix until a dough forms.</li>
        <li>Knead the dough for 5-7 minutes.</li>
      </ol>
    </li>

    <li>
      Let the Dough Rise
      <ol style="list-style-type: lower-alpha; margin-left: 20px">
        <li>Cover the dough with a damp cloth.</li>
        <li>Let it rise for 1 hour, or until doubled in size.</li>
      </ol>
    </li>

    <li>
      Assemble the Pizza
      <ol style="margin-left: 20px">
        <li>Preheat the oven to 475°F (245°C).</li>
        <li>Spread the dough onto a pizza pan.</li>
        <li>Add tomato sauce, cheese, and your favorite toppings.</li>
        <li>Bake for 10-12 minutes or until golden brown.</li>
      </ol>
    </li>
  </ol>

  <!-- No-bullet list for notes -->
  <h2 id="notes">Notes</h2>
  <ul style="list-style-type: none; padding-left: 20px">
    <li>This recipe makes one medium-sized pizza.</li>
    <li>Adjust the toppings and cheese to your taste.</li>
    <li>Try different sauces, like pesto or barbecue, for variety.</li>
  </ul>
